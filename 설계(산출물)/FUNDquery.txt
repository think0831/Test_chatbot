CREATE TABLE FUND(
FUND_NUMBER NUMBER CONSTRAINT FUND_NUMBER_PK PRIMARY KEY,
FUND_NAME VARCHAR2(48) CONSTRAINT FUND_NAME_NN NOT NULL,
FUND_TYPE VARCHAR2(2) CONSTRAINT FUND_TYPE_NN NOT NULL,
BASE_PRICE NUMBER CONSTRAINT FUND_BASE_PRICE_NN NOT NULL,
NAV NUMBER CONSTRAINT FUND_NAV_NN NOT NULL,
TAM VARCHAR2(128) CONSTRAINT FUND_TAM_NN NOT NULL,
FIRST_FEE NUMBER CONSTRAINT FUND_FIRST_FEE_NN NOT NULL,
RESALE_FEE NUMBER CONSTRAINT FUND_RESALE_FEE_NN NOT NULL,
REPURCHASE NUMBER CONSTRAINT FUND_REPURCHASE_NN NOT NULL,
PROFIT NUMBER CONSTRAINT FUND_PROFIT_NN NOT NULL,
TOTAL_PAY NUMBER CONSTRAINT FUND_TOTAL_PAY_NN NOT NULL
);
